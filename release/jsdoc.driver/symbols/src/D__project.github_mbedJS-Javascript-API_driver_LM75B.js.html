1 /**
  2  * @fileOverview LM75Bクラスを定義します。
  3  */
  4 (function(){
  5 var NS=mbedJS;
  6 var MI=MiMicJS;
  7 
  8 
  9 /**
 10  * LM75Bを制御するクラスです。
 11  * <a href="https://github.com/hara41/test_LM75B/tree/master/test_LM75B">test_LM75B</a>と同等の機能を持ちます。
 12  * @constructor
 13  * @name mbedJS.LM75B
 14  * @param {mbedJS.I2C|mbedJS.MCU} i_parent
 15  * インスタンスをバインドするオブジェクトです。MCUの場合はI2Cバスを占有します。
 16  * @param {addr:int]|[addr:int,sda:PinName,scl:PinName]} i_params
 17  * mbedJS.I2Cの場合はI2Cアドレスを指定します。mbedJS.MCUの場合はI2C address,sda,sclの順番で指定します。
 18  * @param {HashMap|Generator|function} i_handler
 19  * 非同期イベントの共通ハンドラの連想配列,Generator,個別コールバック関数の何れかを指定します。
 20  * <p>
 21  * {HashMap} 非同期イベントの共通イベントハンドラです。
 22  * <ul>
 23  * <li>onNew:function() -
 24  * インスタンスが使用可能になった時に呼び出されます。
 25  * </li>
 26  * <li>onRead:function(v) -
 27  * getRead関数のコールバック関数です。
 28  * 	<ul>
 29  * 		<li>v:float - 戻り値です。</li>
 30  * 	</ul>
 31  * </li>
 32  * </ul>
 33  * <p>
 34  * {Generator} Generatorを指定した場合、コールバック関数の引数はyiledの戻り値として取得できます。
 35  * </p>
 36  * <p>
 37  * {function} コールバック関数を指定した場合、RPCが完了したときにonNew相当のコールバック関数が呼び出されます。
 38  * メンバ関数のイベントハンドラは個別に設定してください。
 39  * </p>
 40  * @return {mbedJS.LM75B}
 41  * @example //Callback
 42  * @example //Generator
 43  * @example //Callback hell
 44  */
 45 var CLASS=function LM75B(i_parent,i_params,i_handler)
 46 {
 47 	try{
 48 		var _t=this;
 49 		_t._lc=CLASS;
 50 		var cb=MI._initHandler.call(_t,i_handler);
 51 		MI.assertInt(i_params);		
 52 		//パラメータ退避
 53 		_t._i2c_addr=i_params[0];
 54 		//i2cが準備できたら実行する関数
 55 		function _onNew()
 56 		{
 57 			//control command
 58 			_t._i2c.write(_t._i2c_addr,[0x01,0x00],false,
 59 			function(){
 60 				if(cb){cb();}
 61 				else if(_t._gen){_t._gen.next(_t);}
 62 			});
 63 		}
 64 		//MCUかI2Cか調べる
 65 		if(i_parent.RPC_NS!=NS.I2C.prototype.RPC_NS){
 66 			_t._i2c_attached=true;
 67 			_t._i2c=new NS.I2C(i_parent,[i_params[1],i_params[2]],function(){
 68 				_t._i2c.frequency(10000,function(){
 69 					_onNew();
 70 				});
 71 			});
 72 		}else{
 73 			//I2Cならそのまんま使う
 74 			_t._i2c=i_parent;
 75 			_onNew();
 76 		}
 77 	}catch(e){
 78 		throw new MI.MiMicException(e);
 79 	}	
 80 }
 81 CLASS.prototype=
 82 {
 83 	/** @private */
 84 	RPC_NS:"mbedJS:LM75B",
 85 	/** @private 最後にコールしたAPIです。*/
 86 	_lc:null,
 87 	/** @private Generatorモードの場合にGeneratorオブジェクトを保持します。*/
 88 	_gen:null,
 89 	/** @private コールバック関数の連想配列です。要素はコンストラクタを参照してください。*/
 90 	_event:{},
 91 	_i2c:null,
 92 	_i2c_addr:0,
 93 	_i2c_attached:false,
 94 	/**
 95 	 * Generatorモードのときに使用する関数です。
 96 	 * Generatorモードの時は、yieldと併用してnew LM75B()の完了を待ちます。
 97 	 * @name mbedJS.LM75B#waitForNew
 98 	 * @function
 99 	 */
100 	waitForNew:function LM75B_waitForNew()
101 	{
102 		try{
103 			if(this._lc!=CLASS){throw new MI.MiMicException(MI.Error.NG_ILLEGAL_CALL);}
104 			this._lc=CLASS.waitForNew;
105 		}catch(e){
106 			throw new MI.MiMicException(e);
107 		}
108 	},
109 	/**
110 	 * read相当の関数です。
111 	 * 関数の完了時にonReadイベントが発生します。
112 	 * Generatorモードの時は、yieldと併用して完了を待機できます。
113 	 * @name mbedJS.LM75B#read
114 	 * @function
115 	 * @return　{int}
116 	 * Generatorモードの時はonReadコールバック引数の値を返します。
117 	 */	
118 	read:function LM75B_read()
119 	{
120 		try{
121 			var _t=this;
122 			var cb=MI._getCb(arguments,_t._event.onRead);
123 			MI._assertYield.call(_t);
124 			_t._lc=CLASS.read;			
125 			_t._i2c.write(_t._i2c_addr,[0x00],true,function(){
126 				_t._i2c.read(_t._i2c_addr,2,false,function(v){
127 					var ret = ((v.data[0]<<8)|v.data[1])/256.0;
128 					if(cb){cb(ret);}
129 					else if(_t._gen){_t._gen.next(ret);}
130 					 _t._lc=null;
131 				});
132 			});
133 		}catch(e){
134 			throw new MI.MiMicException(e);
135 		}
136 	},
137 	/**
138 	 * MCUに生成されているオブジェクトを破棄します。
139 	 * @name mbedJS.LM75B#dispose
140 	 * @function
141 	 */
142 	dispose:function LM75B_dispose()
143 	{
144 		try{
145 			var _t=this;
146 			var cb=MI._getCb(arguments,_t._event.onDispose);
147 			MI._assertYield.call(_t);
148 			//自己生成したI2Cの場合のみ削除
149 			if(_t._i2c_attached){
150 				_t._i2c.dispose(function(v){
151 					if(cb){cb(v);}
152 					else if(_t._gen){_t._gen.next(v);}
153 				});
154 			}
155 		}catch(e){
156 			throw new MI.MiMicException(e);
157 		}
158 	}
159 }
160 NS.LM75B=CLASS;
161 }());

  7 /**
  8  * MbedApplicationBoardのPortを制御するクラスです。
  9  * @constructor
 10  * @name mbedAppBoard.Port
 11  * @param {mbedJS.MCU} i_mcu
 12  * インスタンスをバインドするオブジェクトです。
 13  * @param {int} i_ch
 14  * Portのチャンネルを指定します。1,2を指定できます。
 15  * @param {HashMap|Generator|function} i_handler
 16  * 非同期イベントの共通ハンドラの連想配列,Generator,個別コールバック関数の何れかを指定します。
 17  * <p>
 18  * {HashMap} 非同期イベントの共通イベントハンドラです。関数の引数returnは各関数の戻り値です。
 19  * <ul>
 20  * <li>onNew:function() - コンストラクタが完了し、インスタンスが使用可能になった時に呼び出されます。</li>
 21  * <li>onDispose:function() - disposeの完了時に呼び出されます。</li>
 22  * <li>onRead(return) -readの完了時に呼び出されます。</li>
 23  * <li>onRead_u16(return) -read_u16の完了時に呼び出されます。</li>
 24  * </ul>
 25  * </p>
 26  * <p>
 27  * {function} 関数の完了を受け取るコールバック関数です。onNew相当のコールバック関数が呼び出されます。
 28  * </p>
 29  * <p>
 30  * {Generator} yieldコールを行う場合にGeneratorを指定します。
 31  * </p>
 32  * @example //Callback
 33  * @example //Generator
 34  * @example //Callback hell
 35  */
 36 var CLASS=function Port(i_mcu,i_ch,i_handler)
 37 {
 38 	try{
 39 		var _t=this;
 40 		_t._lc=CLASS;
 41 		var cb=MI._initHandler.call(_t,i_handler);
 42 		var pin;
 43 		switch(i_ch)
 44 		{
 45 		case 1:pin=mbedJS.PinName.p19;break;
 46 		case 2:pin=mbedJS.PinName.p20;break;
 47 		default:throw new MI.MiMicException(e);
 48 		}
 49 		_t._ain=new mbedJS.AnalogIn(i_mcu,pin,function(){
 50 			if(cb){cb();}
 51 			else if(_t._gen){_t._gen.next(_t);}
 52 			 _t._lc=null;
 53 		});
 54 		return;
 55 	}catch(e){
 56 		throw new MI.MiMicException(e);
 57 	}	
 58 }

 82 		}

 85 	 * 加速度センサから値を取得します。

 88 	 * @name mbedAppBoard.Port#read_u16
 89 	 * @function
 90 	 * @param {function(return)} i_callback
 91 	 * 省略可能です。関数の完了通知を受け取るコールバック関数を指定します。関数の引数には、return値が入ります。
 92 	 * 省略時はコンストラクタに指定した共通イベントハンドラが呼び出されます。
 93 	 * @return {int}
 94 	 * センサから取得した値の配列です。16bit unsigned int値です。
 95 	 * 戻り値は、コールバック関数、共通コールバック関数、又はyield　returnの何れかで返します。
 96 	 */	
 97 	read_u16:function Port_read_u16()

101 			var cb=MI._getCb(arguments,_t._event.onRead_u16);
102 			MI._assertYield.call(_t);
103 			_t._lc=CLASS.read_u16;
104 			_t._ain.read_u16(function(j){
105 				if(cb){cb(j);}
106 				if(_t._gen){_t._gen.next(j);}
107 				 _t._lc=null;
108 			});
109 		}catch(e){
110 			throw new MI.MiMicException(e);
111 		}
112 	},
113 	/**
114 	 * 加速度センサから値を取得します。
115 	 * 関数の完了時にonRead,又はコールバック関数でイベントを通知します。
116 	 * コンストラクタでGeneratorを指定した場合、yieldと併用して完了を待機できます。
117 	 * @name mbedAppBoard.Port#read
118 	 * @function
119 	 * @param {function(return)} i_callback
120 	 * 省略可能です。関数の完了通知を受け取るコールバック関数を指定します。関数の引数には、return値が入ります。
121 	 * 省略時はコンストラクタに指定した共通イベントハンドラが呼び出されます。
122 	 * @return {float}
123 	 * センサから取得した値の配列です。0<=n<=1の値です。
124 	 * 戻り値は、コールバック関数、共通コールバック関数、又はyield　returnの何れかで返します。
125 	 */

127 	{
128 		try{
129 			var _t=this;
130 			var cb=MI._getCb(arguments,_t._event.onRead);
131 			MI._assertYield.call(_t);
132 			_t._lc=CLASS.read;
133 			_t._ain.read(function(j){
134 				if(cb){cb(j);}
135 				if(_t._gen){_t._gen.next(j);}
136 				 _t._lc=null;
137 			});

139 			throw new MI.MiMicException(e);

146 	 * @name mbedAppBoard.Port#dispose
147 	 * @function
148 	 * @param {function()} i_callback
149 	 * 省略可能です。関数の完了通知を受け取るコールバック関数を指定します。関数の引数には、return値が入ります。
150 	 * 省略時は、コンストラクタに指定した共通イベントハンドラが呼び出されます。
151 	 */

153 	{
154 		try{
155 			var _t=this;
156 			var cb=MI._getCb(arguments,_t._event.onDispose);
157 			MI._assertYield.call(_t);
158 			_t._lc=CLASS.dispose;

160 				if(cb){cb();}
161 				if(_t._gen){_t._gen.next();}
162 				 _t._lc=null;
163 			});
164 		}catch(e){
165 			throw new MI.MiMicException(e);
166 		}
167 	}

  8 /**
  9  * I2Cクラスです。
 10  * <a href="https://mbed.org/handbook/I2C">mbed::I2C</a>と同等の機能を持ちます。
 11  * @constructor
 12  * @name mbedJS.I2C
 13  * @param {mbedJS.Mcu} i_mcu
 14  * インスタンスをバインドするMCUオブジェクトです。
 15  * @param {[PinName,PinName]} i_params
 16  * i2Cバスを構成するピンIDを指定します。sda,sclの順番です。
 17  * @param {HashMap|Generator|function} i_handler
 18  * 非同期イベントの共通ハンドラの連想配列,Generator,個別コールバック関数の何れかを指定します。
 19  * <p>
 20  * {HashMap} 非同期イベントの共通イベントハンドラです。
 21  * <ul>
 22  * <li>onNew:function() -
 23  * インスタンスが使用可能になった時に呼び出されます。
 24  * </li>
 25  * <li>onFrequency:function() -
 26  * frequency関数が完了したときに呼び出されます。
 27  * </li>
 28  * <li>onRead:function(ret,data) -
 29  * read関数が完了したときに呼び出されます。
 30  * <ul>
 31  * <li>ret:int - 成功/失敗フラグを返します。read.1とread.2の場合で意味が異なります。
 32  * read.1の場合、 0:ACK(成功),1:NACK(失敗)です。read.2の場合、読みだした値です。</li>
 33  * <li>data:byte[] - 読みだしたデータの配列です。read.1の場合のみ有効です。</li>
 34  * </ul> 
 35  * </li>
 36  * <li>onWrite:function(ret)-
 37  * write関数が完了したときに呼び出されます。
 38  * <ul>
 39  * <li>ret:int - 成功/失敗フラグを返します。write.1とwrite.2の場合で意味が異なります。
 40  * write.1の場合、ACK:0(成功),NACK:それ以外です。write.2の場合、ACKを受信すると1を返します。</li>
 41  * </ul> 
 42  * </li>
 43  * <li>onStart:function() -
 44  * start関数が完了したときに呼び出されます。
 45  * </li>
 46  * <li>onStop:function() -
 47  * stop関数が完了したときに呼び出されます。
 48  * </li>
 49  * </ul>
 50  * <p>
 51  * {Generator} Generatorを指定した場合、コールバック関数の引数はyiledの戻り値として取得できます。
 52  * </p>
 53  * <p>
 54  * {function} コールバック関数を指定した場合、RPCが完了したときにonNew相当のコールバック関数が呼び出されます。
 55  * メンバ関数のイベントハンドラは個別に設定してください。
 56  * </p>
 57  * @return {mbedJS.I2C}
 58  * @example //Callback
 59  * var st=0;
 60  * var mcu=new mbedJS.Mcu("192.168.128.39",
 61  * {
 62  *   onNew:function(){
 63  *     var i2c=new mbedJS.I2C(mcu,[mbedJS.PinName.p28,mbedJS.PinName.p27],{
 64  *     onNew:function(){
 65  *       i2c.frequency(100000);
 66  *     },
 67  *     onFrequency:function()
 68  *     {
 69  *       i2c.start();
 70  *     },
 71  *     onStart:function(){
 72  *       st=0;
 73  *       i2c.write(1);
 74  *     },
 75  *     onWrite:function(v){
 76  *       if(st==0){
 77  *         i2c.write(0,[1,2,3],false);
 78  *         st++;
 79  *       }else{
 80  *         i2c.read(1);
 81  *         st=0;
 82  *       }
 83  *     },
 84  *     onRead:function(v){
 85  *       if(st==0){
 86  *         i2c.read(1,2,false);
 87  *         st++;
 88  *       }else{
 89  *         i2c.stop();
 90  *         }
 91  *     },
 92  *     onStop:function(){
 93  *       mcu.close();
 94  *     }
 95  *     });
 96  *   },
 97  *   onClose:function(){
 98  *   },
 99  *   onError:function(){
100  *     alert("Error");
101  *   }
102  * });
103  * @example //Generator
104  * var g=function*(){
105  * try{
106  *   var v;
107  *   var mcu=new mbedJS.Mcu("192.168.128.39",g);
108  *   yield mcu.waitForNew();
109  *   var i2c=new mbedJS.I2C(mcu,[mbedJS.PinName.p28,mbedJS.PinName.p27],g);
110  *   yield i2c.waitForNew();
111  *   yield i2c.frequency(100000);
112  *   yield i2c.start();
113  *   yield i2c.write(1);
114  *   yield i2c.write(0,[1,2,3],false);
115  *   yield i2c.read(1);
116  *   yield i2c.read(1,2,false);
117  *   yield i2c.stop();
118  *   yield mcu.close();
119  * }catch(e){
120  *   mcu.shutdown();
121  *   alert(e);
122  *   throw e;
123  * }
124  * }();
125  * g.next();
126  */
127 var CLASS=function I2C(i_mcu,i_params,i_handler)
128 {
129 	try{
130 		var _t=this;
131 		_t._mcu=i_mcu;
132 		_t._lc=CLASS;
133 		var cb=MI._initHandler.call(_t,i_handler);
134 		MI.assertInt(i_params);		
135 		_t._mcu.rpc(_t.RPC_NS+":_new1",i_params[0]+","+i_params[1],
136 			function (j)
137 			{
138 				_t._oid=j.result[0];
139 				if(cb){cb();}
140 				if(_t._gen){_t._gen.next(_t);}
141 				_t._lc=null;
142 			}
143 		);
144 	}catch(e){
145 		throw new MI.MiMicException(e);
146 	}	
147 }
148 CLASS.prototype=
149 {
150 	/** @private */
151 	RPC_NS:"mbedJS:I2C",
152 	/** @private 最後にコールしたAPIです。*/
153 	_lc:null,
154 	/** @private Generatorモードの場合にGeneratorオブジェクトを保持します。*/
155 	_gen:null,
156 	/** @private コールバック関数の連想配列です。要素はコンストラクタを参照してください。*/
157 	_event:{},
158 	/** @private リモートインスタンスのオブジェクトIDです。*/
159 	_oid:null,
160 	/**
161 	 * Generatorモードのときに使用する関数です。
162 	 * Generatorモードの時は、yieldと併用してnew I2C()の完了を待ちます。
163 	 * @name mbedJS.I2C#waitForNew
164 	 * @function
165 	 */
166 	waitForNew:function I2C_waitForNew()
167 	{
168 		try{
169 			if(this._lc!=CLASS){throw new MI.MiMicException(MI.Error.NG_ILLEGAL_CALL);}
170 			this._lc=CLASS.waitForNew;
171 		}catch(e){
172 			throw new MI.MiMicException(e);
173 		}
174 	},
175 	/**
176 	 * Hz単位でバスの速度を指定します。
177 	 * 関数の完了時にonFrequencyイベントが発生します。
178 	 * Generatorモードの時は、yieldと併用して完了を待機できます。
179 	 * @name mbedJS.I2C#frequency
180 	 * @function
181 	 * @param {int} i_hz
182 	 * Hz単位のバス速度です。
183 	 * @return　{int}
184 	 * Callbackモードの時はRPCメソッドのインデクスを返します。
185 	 */
186 	frequency:function I2C_frequency(i_hz)
187 	{
188 		try{
189 			var _t=this;
190 			var cb=MI._getCb(arguments,_t._event.onFrequency);
191 			MI._assertYield.call(_t);
192 			_t._lc=CLASS.frequency;
193 			MI.assertInt(i_hz);
194 			return _t._mcu.rpc(_t.RPC_NS+":frequency",_t._oid+","+i_hz,
195 				function (j)
196 				{
197 					if(cb){cb();}
198 					if(_t._gen){_t._gen.next();}
199 					 _t._lc=null;
200 				}
201 			);
202 		}catch(e){
203 			throw new MI.MiMicException(e);
204 		}			
205 	},
206 	/**
207 	 * 引数が3個の場合
208 	 * @name mbedJS.I2C#read:1
209 	 * @function
210 	 * @param {int} address
211 	 * 8ビットのI2CSlaveアドレスです。
212 	 * @param {int} length
213 	 * 読み出すデータの長さです。256未満の値を指定してください。
214 	 * @param {boolean} repeated
215 	 * Repeated start, true - do not send stop at end
216 	 * Optionalです。省略時はfalseです。
217 	 * @return {int}
218 	 * Callbackモードの時はRPCメソッドのインデクスを返します。
219 	 * @return {HashMap}
220 	 * {ret:int,data:[byte]}
221 	 * Generatorの場合は戻り値オブジェクトを返します。
222 	 * <ul>
223 	 * <li>ret:int - 成功フラグ 0:ACK(成功),1:NACK(失敗)</li>
224 	 * <li>data:byte[] - 読みだしたデータ</li>
225 	 * </ul>
226 	 */
227 	/**
228 	 * 引数が1個の場合
229 	 * @name mbedJS.I2C#read:2
230 	 * @function
231 	 * @param {int} ack
232 	 * indicates if the byte is to be acknowledged (1 = acknowledge)
233 	 * @return {int}
234 	 * Callbackモードの時はRPCメソッドのインデクスを返します。
235 	 * @return {int}
236 	 * Generatorモードの時は読みだした値を返します。
237 	 */
238 	/**
239 	 * バスから値を読み出します。
240 	 * 関数の完了時にonReadイベントが発生します。
241 	 * Generatorモードの時は、yieldと併用して完了を待機できます。
242 	 * @name mbedJS.I2C#read
243 	 * @function
244 	 * @param ...
245 	 * 詳細はmbedJS.I2C#read:Nを参照してください。
246 	 */	
247 	read:function I2C_read(/*...*/)
248 	{
249 		try{
250 			var _t=this;
251 			var cb=MI._getCb(arguments,_t._event.onRead);
252 			MI._assertYield.call(_t);
253 			_t._lc=CLASS.read;
254 			function rcb(j){
255 				var v=j.result.length>1?{ret:j.result[0],data:MI.bstr2byteArray(j.result[1])}:j.result[0];
256 				if(cb){cb(v);}
257 				if(_t._gen){_t._gen.next(v);}
258 				 _t._lc=null;
259 			}
260 			//ベース引数の数で処理の切り替え
261 			if(MI._getBaseArgsLen(arguments)==1){
262 				MI.assertInt(arguments[0]);
263 				return _t._mcu.rpc(_t.RPC_NS+":read2",_t._oid+","+arguments[0],rcb);
264 			}else{
265 				var a=arguments;
266 				MI.assertInt([a[0],a[1]]);
267 				return _t._mcu.rpc(_t.RPC_NS+":read1",_t._oid+","+a[0]+","+a[1]+","+(a[2]?1:0),rcb);
268 			}
269 		}catch(e){
270 			throw new MI.MiMicException(e);
271 		}
272 	},
273 	/**
274 	 * 引数が3個の場合
275 	 * @name mbedJS.I2C#write:1
276 	 * @function
277 	 * @param {int} address
278 	 * 8ビットのI2CSlaveアドレスです。
279 	 * @param {byte[]} data
280 	 * 送信するデータを格納したバイト配列です。
281 	 * @param {boolean} repeated
282 	 * Repeated start, true - do not send stop at end
283 	 * Optionalです。省略時はfalseです。
284 	 * @return {int}
285 	 * Callbackモードの時はRPCメソッドのインデクスを返します。
286 	 * @return {int}
287 	 * Generatorモードの時は成功/失敗フラグを返します。ACK:0(成功),NACK:それ以外です。
288 	 */
289 	/**
290 	 * 引数が1個の場合
291 	 * @name mbedJS.I2C#write:2
292 	 * @function
293 	 * @param {int} data
294 	 * 送信データを指定します。
295 	 * @return {int}
296 	 * Callbackモードの時はRPCメソッドのインデクスを返します。
297 	 * @return {int}
298 	 * Generatorモードの時は成功/失敗フラグを返します。ACKを受信すると1を返します。
299 	 */	
300 	/**
301 	 * バスに値を書き込みます。
302 	 * 関数の完了時にonWriteイベントが発生します。
303 	 * Generatorモードの時は、yieldと併用して完了を待機できます。
304 	 * @name mbedJS.I2C#write
305 	 * @function
306 	 * @param ...
307 	 * 詳細はmbedJS.I2C#write:Nを参照してください。
308 	 */	
309 	write:function I2C_write(/*...*/)
310 	{
311 		try{
312 			var _t=this;
313 			var cb=MI._getCb(arguments,_t._event.onWrite);
314 			MI._assertYield.call(_t);
315 			_t._lc=CLASS.write;
316 			function rcb(j){
317 				var v=j.result[0];
318 				if(cb){cb(v);}
319 				if(_t._gen){_t._gen.next(v);}
320 				 _t._lc=null;
321 			}
322 			if(MI._getBaseArgsLen(arguments)==1){
323 				MI.assertInt(arguments[0]);
324 				return _t._mcu.rpc(_t.RPC_NS+":write2",_t._oid+","+arguments[0],rcb);
325 			}else{
326 				var a=arguments;
327 				MI.assertInt(a[0]);
328 				return _t._mcu.rpc(_t.RPC_NS+":write1",_t._oid+","+a[0]+",\""+MI.byteArray2bstr(a[1])+"\","+(a[2]?1:0),rcb);
329 			}
330 		}catch(e){
331 			throw new MI.MiMicException(e);
332 		}			
333 	},
334 	/**
335 	 * I2Cバスを開始状態にします。
336 	 * 関数の完了時にonStartイベントが発生します。
337 	 * Generatorモードの時は、yieldと併用して完了を待機できます。
338 	 * @name mbedJS.I2C#start
339 	 * @function
340 	 * @return　{int}
341 	 * Callbackモードの時はRPCメソッドのインデクスを返します。
342 	 */
343 	start:function I2C_start()
344 	{
345 		try{
346 			var _t=this;
347 			var cb=MI._getCb(arguments,_t._event.onStart);
348 			MI._assertYield.call(_t);
349 			_t._lc=CLASS.start;			
350 			return _t._mcu.rpc(_t.RPC_NS+":start",_t._oid,
351 				function (j)
352 				{
353 					if(cb){cb();}
354 					if(_t._gen){_t._gen.next();}
355 					 _t._lc=null;
356 				}
357 			);
358 		}catch(e){
359 			throw new MI.MiMicException(e);
360 		}			
361 	},
362 	/**
363 	 * I2Cバスを停止状態にします。
364 	 * 関数の完了時にonStopイベントが発生します。
365 	 * Generatorモードの時は、yieldと併用して完了を待機できます。
366 	 * @name mbedJS.I2C#stop
367 	 * @function
368 	 * @return　{int}
369 	 * Callbackモードの時はRPCメソッドのインデクスを返します。
370 	 */
371 	stop:function I2C_stop()
372 	{
373 		try{
374 			var _t=this;
375 			var cb=MI._getCb(arguments,_t._event.onStop);
376 			MI._assertYield.call(_t);
377 			_t._lc=CLASS.stop;
378 			return _t._mcu.rpc(_t.RPC_NS+":stop",_t._oid,
379 				function (j)
380 				{
381 					if(cb){cb();}
382 					if(_t._gen){_t._gen.next();}
383 					 _t._lc=null;
384 				}
385 			);
386 		}catch(e){
387 			throw new MI.MiMicException(e);
388 		}			
389 	},
390 	/**
391 	 * MCUに生成されているオブジェクトを破棄します。
392 	 * @name mbedJS.I2C#dispose
393 	 * @function
394 	 */

396 	{
397 		return this._mcu._dispose.apply(this,arguments);
398 	}
